SELECT 
    p.ID as booking_id,
    MAX(CASE WHEN pm.meta_key = 'chbs_form_field_firstname' THEN pm.meta_value END) as first_name,
    MAX(CASE WHEN pm.meta_key = 'chbs_form_field_lastname' THEN pm.meta_value END) as last_name,
    MAX(CASE WHEN pm.meta_key = 'chbs_form_field_email' THEN pm.meta_value END) as email,
    MAX(CASE WHEN pm.meta_key = 'chbs_form_field_phone' THEN pm.meta_value END) as phone,
    MAX(CASE WHEN pm.meta_key = 'chbs_booking_route_pickup' OR pm.meta_key = 'chbs_location_pickup' THEN pm.meta_value END) as pickup_address,
    MAX(CASE WHEN pm.meta_key = 'chbs_booking_route_destination' OR pm.meta_key = 'chbs_location_destination' THEN pm.meta_value END) as dropoff_address,
    MAX(CASE WHEN pm.meta_key = 'chbs_booking_date_pickup' THEN pm.meta_value END) as pickup_date,
    MAX(CASE WHEN pm.meta_key = 'chbs_booking_time_pickup' THEN pm.meta_value END) as pickup_time,
    MAX(CASE WHEN pm.meta_key = 'chbs_booking_date_return' THEN pm.meta_value END) as return_date,
    MAX(CASE WHEN pm.meta_key = 'chbs_booking_time_return' THEN pm.meta_value END) as return_time,
    MAX(CASE WHEN pm.meta_key = 'chbs_passenger_adult_number' THEN pm.meta_value END) as passengers,
    MAX(CASE WHEN pm.meta_key = 'chbs_booking_price_total' THEN pm.meta_value END) as total_price,
    MAX(CASE WHEN pm.meta_key = 'chbs_form_field_flight_departure_no' OR pm.meta_key = 'chbs_form_field_flightno' THEN pm.meta_value END) as flight_number,
    MAX(CASE WHEN pm.meta_key = 'chbs_form_field_flight_departure_time' THEN pm.meta_value END) as flight_time
FROM wp_posts p
LEFT JOIN wp_postmeta pm ON p.ID = pm.post_id
WHERE p.post_type = 'chbs_booking' 
    AND p.post_status = 'publish'
GROUP BY p.ID
ORDER BY pickup_date DESC;
